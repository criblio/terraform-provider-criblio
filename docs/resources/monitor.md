# criblio_monitor (Resource)

Manages an Aetos monitor (`IAetosMonitorConf`) via `POST/GET/PATCH/DELETE /products/aetos/monitors[/{id}]`.

## Example Usage

```terraform
resource "criblio_monitor" "example" {
  id      = "tf-my-monitor"
  name    = "My Terraform Monitor"
  enabled = true
  type    = "threshold"

  priority = jsonencode({ value = "P2" })
  team     = jsonencode({ value = "ops" })

  query = jsonencode({
    A = { mode = "promql", promql = "up" }
  })

  expr = jsonencode([])

  firing_condition = jsonencode({
    fire_delay  = 300
    clear_delay = 60
  })

  firing_rule = jsonencode({
    label     = "down"
    threshold = [
      {
        severity     = "critical"
        limit        = 0
        operator     = "lt"
        includedTags = []
        excludedTags = []
      }
    ]
  })

  metadata     = jsonencode({})
  notification = jsonencode({ enabled = false, type = "policy", config = [] })
  silence      = []

  description = "Monitors whether hosts are up."
  dataset_id  = "metrics"
}
```
